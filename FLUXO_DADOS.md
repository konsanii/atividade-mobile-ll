Começando...');\n  \n  // Requisição começa mas NÃO ESPERA\n  const dados = fetch('url');\n  console.log('2. Isso imprime ANTES da resposta!'); // Imprime antes\n  console.log('3. Começando...');\n  \n  // Requisição começa e ESPERA\n  const dados = await fetch('url');\n  console.log('2. Agora imprime DEPOIS da resposta'); // Espera\n  console.log('3. Dados:', dados); // Preenchido ✅\n}\n\n// Resultado: 1 → (espera) → 2 → 3\n```\n\n---\n\n## ⏱️ Timeline: O que acontece quando chama uma função async\n\n```\nTEMPO: 0ms\n├─ Começar handleBuscarCep()\n├─ setCarregando(true)\n├─ Chamar buscarEnderecoPorCep()\n│\nTEMPO: 50ms\n├─ await fetch(...) - PAUSA\n├ 🕐 AGUARDA RESPOSTA\n│\nTEMPO: 500ms - 2000ms (network latency)\n├─ 🌐 RESPOSTA CHEGA\n├─ await response.json() - PAUSA\n│ 🕐 AGUARDA CONVERSÃO\n│\nTEMPO: 2010ms\n├─ ✅ Dados prontos\n├─ setRua(endereco.rua)\n├─ setBairro(endereco.bairro)\n├─ setCidade(endereco.cidade)\n├─ setEstado(endereco.estado)\n├─ setCepBuscado(true)\n├─ setCarregando(false)\n│\nTEMPO: 2015ms\n└─ TELA ATUALIZA COM VALORES\n```\n\n**Observação**: Se não usasse `await`, o código continuaria em 50ms\ne tentaria usar dados que ainda não chegaram (undefined)!\n\n---\n\n## 🎯 Padrão de Tratamento de Erros\n\n```javascript\nconst buscarComSeguranca = async (cep) => {\n  try {\n    // 🟢 TRY: Tenta fazer isso\n    console.log('Buscando...');\n    \n    const endereco = await buscarEnderecoPorCep(cep);\n    // Se algo der errado aqui, pula para CATCH\n    \n    console.log('Sucesso!');\n    return endereco;\n    \n  } catch (erro) {\n    // 🔴 CATCH: Se deu erro, vem pra cá\n    console.error('Erro:', erro.message);\n    \n    if (erro.message.includes('CEP não encontrado')) {\n      Alert.alert('Erro', 'CEP inválido, tente outro');\n    } else if (erro.message.includes('Network')) {\n      Alert.alert('Erro', 'Sem conexão de internet');\n    } else {\n      Alert.alert('Erro', 'Algo deu errado');
